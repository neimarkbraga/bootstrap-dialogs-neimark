Bootstrap Dialogs Neimark v1.2.1 Demo

Dependencies:

  • jquery @ ^3.2.1
  • popper.js @ ^1.11.0
  • bootstrap @ ^4.0.0-beta

Initialization: 

    var myDialog = new BootstrapDialog({
        preloaderClass: 'bg-dark',
        buttonClass: 'btn-primary',
        dialogTitleTag: 'h4',
        dialogSize: '',
        animation: true
    });

Alert Dialog Code: 

    myDialog.alert('Alert Demo', 'Some message', function () {
        console.log('Alert Callback Executed');
    });

Confirm Dialog Code: 

    myDialog.confirm('Confirm Demo', 'Some message', function (result) {
        console.log('Confirm Callback Executed. Result: ' + result);
    });

Prompt Dialog Code: 

    var promptOptions = {
        label: 'Name',
        type: 'text',
        placeholder: 'Enter your name',
        value: '',
        list: 'myList',
        required: false
    };

    myDialog.prompt('Prompt Demo', 'Some message', function (result) {
        console.log('Prompt Callback Executed. Result: ' + result);
    }, promptOptions);

Preloader Dialog Code: 

    var loader = new myDialog.preloader();

    setTimeout(function () {
        loader.changeLabel('Hello World!');
        setTimeout(function () {
            loader.destroy();
        }, 2000);
    }, 2000);
